Canvas c = Canvas.getCanvas();

int height = c.getHeight();
int width = c.getWidth();

int xPos = 0;
int yPos = 0;

int currentXPos = 0;
int currentYPos = 0;

java.util.Random rand = new java.util.Random();

int count = 0;

Circle circle1 = new Circle();
Circle circle2 = new Circle();

circle1.makeVisible();
circle2.changeColor("red");
circle2.moveRight();
circle2.makeVisible();

while (count <= 10) {
	
	String[] colours = {"red","blue","yellow","green","gray","black"};

	circle1.slowMoveHorizontal(rand.nextInt(width) - circle1.getXCoord());
	circle1.slowMoveVertical(rand.nextInt(height) - circle1.getYCoord());

	boolean flag = false;
	while (flag != true){

		int randomNumber = rand.nextInt(colours.length);

		if (circle1.getColor() != colours[randomNumber]){
			circle1.changeColor(colours[randomNumber]);
			flag = true;
		}
	}
	count++;
}
